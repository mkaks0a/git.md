void main() {

  int age = 29;
  //誕生年の取得関数呼び出し
  int birthYear = getBirthYear(year, age);

  //うるう年判定関数呼び出し
  if (isLearpYear(birthYear)) {
    print("うるう年です");
  } else {
    print("うるう年ではありません");
  }
}

int getBirthYear(int year, int age) {
  return year - age;
}

bool isLearpYear(int year) {
  if (year % 400 == 0) return true;
  if (year % 100 == 0) return false;
  if (year % 4 == 0) return true;
  return false;
}

ポイント7：ループで条件分岐
総まとめ：問題
   たかしくんは秒速90cmで家から学校へ向かう。
   ひろしくんは秒速50cmで家から公園に向かう。
   あいこさんは秒速5cmで家からコンビニに向かう。
   そのまま、目的地で、かかった時間の2倍たいざいし、
   生きの半分の速度で家に戻る。最も早く家につくのは？
    家 ~ コンビニ ~ 公園 ~ 学校     直線状
     1km        0.5km  1km 
     void main() {
  // 目的地まで    単位 mm/sに
  const double school = 1 * 1000 * 100; // 1km
  const double park = 0.5 * 1000 * 100; // 0.5km
  const double shoopstor = 1 * 1000 * 100; // 1km
  //   速度
  const double takashiSpeed = 90;
  const double hiroshiSpeed = 50;
  const double aicoSpeed = 5;

  double totalTime(double time, double speed) {
    double gatTime = time / speed;
    double stayTime = gatTime * 2;
    double turnSpeed = speed / 2;
    double turnTime = time / turnSpeed;
    return gatTime + stayTime + turnTime;
  }

  double takashiTime = totalTime(school, takashiSpeed);
  double hiroshiTime = totalTime(park, hiroshiSpeed);
  double aicoTime = totalTime(shoopstor, aicoSpeed);
  if (takashiTime < hiroshiTime && takashiTime < aicoTime) {
    print("たかしくん");
  } else if (hiroshiTime < aicoTime) {
    print("ひろしくん");
  } else {
    print("あいこさん");
  }
}
